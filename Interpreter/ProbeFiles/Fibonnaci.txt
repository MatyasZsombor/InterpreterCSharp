let fibonacci = fn(x)
{
  if(x == 0 || x == 1)
  {
    return x;
  }
  else
  {
    return fibonacci(x - 1) + fibonacci(x - 2); 
  }
};

let y = get();
let fib = fibonacci(y - 1);
put("The " + string(y) + ". Fibonacci number is: " + string(fib));
write("/Users/matyaszsombor/Projects/Interpreter/fib.txt", fib);
